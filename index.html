mkdir new, rm new, rm -rf new - створення файлів, папок

ls - список папок

cd 'folder-name' - перейти до папки

git - показує список доступних команд

git init - створює папку, яка буде слідкувати за змінами в теці

got status - перевіряє зміни

git add - додає файли до перевірки

git commit -m ' ' - опис змін
